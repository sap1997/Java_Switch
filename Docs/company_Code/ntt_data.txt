String s = "aahbbfjfh";

Set<String> uniquechars = new HashSet<>();

uniquechars = s.chars().mapToObj(c->(char)c).collect(Collectors.groupingBy(c->c,collectors.counting()).entrySet()
		.stream().filter(e->e.getValue()==1).map(e->e.getKey()).collect(Collectors.toSet());


Map<Character,Integer> charcount = new HashMap<>();
Set<String> uniquechars = new HashSet<>();


for(char c : s.toCharArray())
{
	charcount.put(c,charcount.getOrDefault(c,0)+1));

}

for(Map.Entry<Character,Integer> kv : charcount.entrySet())
{
	if(kv.getValue()==1)
	{
		uniquechars.add(kv.getKey())
	}
